<!DOCTYPE html>

<html lang="en">
	<head>
		<title>Pizza Price</title>		
		
		<script>
			function checkName()
			{				
				//make sure a name is entered, other fields already have prechosen selections
				if (document.pizzaForm.userName.value == "" || document.pizzaForm.userName.value == null)
				{
					alert("Please enter your name!");
				
				}
				
				else
				{
					//variables needed
					var items = document.getElementsByName('toppings');
					var extras = "";
					var price = 0;
					
					//set price to pizza size selected
					switch(pizzaForm.size.value)
					{
						case "large":
							price = 15;
							break;
						
						case "medium":
							price = 12;
							break;
							
						case "small":
							price = 9;
							break;
							
						case "personal":
							price = 6;
							break;
					
					}
					
					//checks which toppings were checked and throws in extras 
					for(x = 0; x < items.length; x++)
					{
						if(items[x].checked)
						{
							extras += (items[x].value + "\n");
							price++; //adds $1 for each topping
						
						}
					
					}
					
					//standard script if no toppings added
					var confScript = "Thanks for the order, " + pizzaForm.userName.value  + "\n\nYour total is $" + price
							+ "\n\nYour " + pizzaForm.size.value + " " + pizzaForm.crust.value + " crust pizza will be ready soon!";
					
					//if toppings were added, add them to script
					if(extras != "")
						confScript += "\n\nAdditional toppings selected:\n" + extras;
					
					//display script
					alert(confScript);
				
				}
				
			}

			
		</script>
	</head>	
	<body>

		<form name = "pizzaForm" onsubmit = "checkName()">

			Personal ($6): <input type = "radio" name = "size" value = "personal"/>
